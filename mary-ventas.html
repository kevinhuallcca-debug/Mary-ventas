function buildSidebar(rol,nombre){
  var sb=document.getElementById('sidebar');
  if(!sb)return;
  var logoEl=document.querySelector('#app .logo img');
  var logoSrc=logoEl?logoEl.src:'';
  var items=[];
  if(rol==='admin'){
    items=[
      ['dashboard','📊','Resumen'],
      ['nueva','➕','Nueva venta'],
      ['historial','📋','Historial'],
      ['personal','👥','Personal'],
      ['productos','🍗','Productos'],
      ['gastos','💸','Gastos'],
      ['sesiones','🔒','Sesiones']
    ];
  } else if(rol==='cajero'){
    items=[['dashboard','📊','Resumen'],['nueva','➕','Nueva venta'],['gastos','💸','Gastos']];
  }
  var menuHTML='<div class="sb-menu">';
  items.forEach(function(it){
    menuHTML+='<button class="sb-btn'+(activeTab===it[0]?' on':'')+'" onclick="showTab(\''+it[0]+'\')"><span class="sb-icon">'+it[1]+'</span>'+it[2]+'</button>';
  });
  menuHTML+='</div>';
  sb.innerHTML=
    '<div class="sb-logo"><img src="'+logoSrc+'" alt="Mary"></div>'+
    '<div class="sb-role">'+(rol==='admin'?'Administrador':'Cajero · '+nombre)+'</div>'+
    menuHTML+
    '<div class="sb-bottom"><button class="sb-logout" onclick="logout()"><span>🚪</span> Cerrar sesión</button></div>';
  sb.style.display='flex';
}
function syncSidebar(name){
  document.querySelectorAll('#sidebar .sb-btn').forEach(function(btn){
    var oc=btn.getAttribute('onclick')||'';
    btn.classList.toggle('on',oc.indexOf("'"+name+"'")>-1);
  });
}
